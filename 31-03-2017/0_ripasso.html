      var demo_2_spans = document.getElementsByClassName( "demo_2" );
      var colors       = [ "red", "green", "blue", "yellow", "purple" ];
      for ( var i = 0; i < demo_2_spans.length; i++ ) {
        var demo_2_span         = demo_2_spans[ i ];
        var rand                = Math.floor( Math.random() * colors.length );
        demo_2_span.style.color = colors[ rand ];
      }
