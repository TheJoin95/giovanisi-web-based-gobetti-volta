      var demo_2_spans = document.getElementsByClassName( "demo_2" );
      var colors       = [ "red", "green", "blue", "yellow", "purple" ];
      for ( var i = 0; i < demo_2_spans.length; i++ ) {
        var demo_2_span         = demo_2_spans[ i ];
        var rand                = Math.floor( Math.random() * colors.length );
        demo_2_span.style.color = colors[ rand ];
      }

      function demo_4_button_fratello_onclick( event ) {
        var msg = "";
        msg += "event.clientX " + event.clientX;
        msg += "<br/>";
        msg += "event.clientY " + event.clientY;
        msg += "<br/>";
        msg += "event.screenX " + event.screenX;
        msg += "<br/>";
        msg += "event.screenY " + event.screenY;
        msg += "<br/>";
        msg += "event.target " + event.target.toString();

        var demo_4_span = document.getElementById("demo_4_span");
        demo_4_span.innerHTML = msg;

        var demo_4_p = document.getElementById("demo_4_p");
        demo_4_p.style.display = "block";
      }
